    字符串分割¶

    代码内容¶

    #include <iostream>
#include <string>
#include <vector>
using std::vector;
using std::string;

// 按照分割字符分割字符串
void split(const string &str, const string &delimiters, vector<string> &elems, bool skip_empty = true) 
{
    elems.clear();
    string::size_type pos, prev = 0;
    while ((pos = str.find_first_of(delimiters, prev)) != string::npos) 
    {
        if (pos > prev) 
        {
            if (skip_empty && 1 == pos - prev) 
                break;

            elems.emplace_back(str, prev, pos - prev);
        }
        prev = pos + 1;
    }
    if (prev < str.size()) 
        elems.emplace_back(str, prev, str.size() - prev);
}

    函数相关参数¶

    void split(const string &str, const string &delimiters, vector<string> &elems, bool skip_empty = true);
/* 
str:            用于分割的字符串
delimiters:     分割字符（ASCII）组成的字符串
elems:          分割结果
skip_empty:     是否跳过空格（若为true，则不以空格为分割条件）
*/

    函数调用实例¶

    vector<string> result_s;
split("日期 时间:2020/02/05-08-06-59", "/-:", result_s);

    运行结束后，result_s的值如下：

    下标 内容
    [0] "日期 时间"
    [1] "2020"
    [2] "02"
    [3] "05"
    [4] "08"
    [5] "06"
    [6] "59"
